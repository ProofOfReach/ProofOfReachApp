ProofOfReach SDK Documentation

The ProofOfReach SDK provides a simple way to monetize your content with Bitcoin's Lightning Network. It's designed to be easy to integrate while providing powerful features for publishers.

Quick Start

// 1. Add the SDK to your site
<script src="https://cdn.proofofreach.com/sdk/v1.js"></script>

// 2. Create a container for the ad
<div id="my-ad-container"></div>

// 3. Initialize and render an ad
<script>
  window.addEventListener("load", () => {
    ProofOfReachSDK.renderAd("my-ad-container", {
      pubkey: "YOUR_NOSTR_PUBKEY", // Your Nostr public key
      adType: "standard",          // standard, small, or banner
      testMode: false,             // Set to true for testing
    });
  });
</script>

Configuration Options

ProofOfReachSDK.renderAd("my-ad-container", {
  // Required parameters
  pubkey: "YOUR_NOSTR_PUBKEY", // Your Nostr public key
  
  // Optional parameters
  adType: "standard",          // standard, small, or banner
  testMode: false,             // Enable test mode (no real payments)
  refreshInterval: 300000,     // Refresh ad every 5 minutes (in ms)
  defaultStyles: true,         // Apply default container styles
  
  // Event callbacks
  onPayment: (sats) => {       // Called when payment is received
    console.log(`Received ${sats} satoshis!`);
  },
  onError: (error) => {        // Called when error occurs
    console.error(`Ad error: ${error}`);
  },
  onLoad: () => {              // Called when ad loads successfully
    console.log("Ad loaded successfully");
  }
});

Integration Examples

WordPress Plugin

Add the following code to your WordPress theme's functions.php file: 

function add_proof_of_reach_sdk() {
  echo '<script src="https://cdn.proofofreach.com/sdk/v1.js"></script>';
  echo '<script>
    window.addEventListener("load", () => {
      // Add ad to sidebar
      if (document.getElementById("sidebar")) {
        ProofOfReachSDK.renderAd("sidebar", {
          pubkey: "YOUR_NOSTR_PUBKEY",
          adType: "small"
        });
      }
      
      // Add ad after first paragraph
      const paragraphs = document.querySelectorAll(".entry-content p");
      if (paragraphs.length > 0) {
        const adContainer = document.createElement("div");
        adContainer.id = "por-content-ad";
        paragraphs[0].after(adContainer);
        
        ProofOfReachSDK.renderAd("por-content-ad", {
          pubkey: "YOUR_NOSTR_PUBKEY",
          adType: "standard"
        });
      }
    });
  </script>';
}
add_action("wp_footer", "add_proof_of_reach_sdk");

React Component

Create a reusable React component for ProofOfReach ads: 

import { useEffect, useRef } from 'react';

function ProofOfReachAd({ adType = 'standard', className = '' }) {
  const adRef = useRef(null);
  const adInstanceRef = useRef(null);
  
  useEffect(() => {
    // Load the SDK script dynamically
    const script = document.createElement('script');
    script.src = 'https://cdn.proofofreach.com/sdk/v1.js';
    script.async = true;
    
    script.onload = () => {
      if (adRef.current && window.ProofOfReachSDK) {
        // Render the ad
        adInstanceRef.current = window.ProofOfReachSDK.renderAd(adRef.current.id, {
          pubkey: process.env.REACT_APP_NOSTR_PUBKEY,
          adType,
          testMode: process.env.NODE_ENV !== 'production'
        });
      }
    };
    
    document.body.appendChild(script);
    
    return () => {
      // Clean up on unmount
      if (adInstanceRef.current) {
        adInstanceRef.current.remove();
      }
    };
  }, [adType]);
  
  return (
    <div 
      id={`por-ad-${Math.random().toString(36).substring(2, 9)}`} 
      ref={adRef}
      className={className}
    />
  );
}

export default ProofOfReachAd;

Next.js Integration

For Next.js sites, create a dynamic component with proper SSR handling: 

// components/ProofOfReachAd.js
import { useEffect, useRef } from 'react';

function ProofOfReachAd({ adType = 'standard' }) {
  const adRef = useRef(null);
  
  useEffect(() => {
    // Load SDK and render ad on client-side only
    const loadSDK = async () => {
      if (window.ProofOfReachSDK) return;
      
      await new Promise((resolve) => {
        const script = document.createElement('script');
        script.src = 'https://cdn.proofofreach.com/sdk/v1.js';
        script.async = true;
        script.onload = resolve;
        document.body.appendChild(script);
      });
      
      if (adRef.current) {
        window.ProofOfReachSDK.renderAd(adRef.current.id, {
          pubkey: process.env.NEXT_PUBLIC_NOSTR_PUBKEY,
          adType,
          testMode: process.env.NODE_ENV !== 'production'
        });
      }
    };
    
    loadSDK();
  }, [adType]);
  
  return (
    <div
      id={`por-ad-${Math.random().toString(36).substring(2, 9)}`}
      ref={adRef}
      style={{ minHeight: '150px' }}
    />
  );
}

export default ProofOfReachAd;
